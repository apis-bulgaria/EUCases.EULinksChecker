358 /** @private */
359 function replaceSelection(html, selectInserted) {
360     var sel, range, fragment;
361 
362     if (typeof window.getSelection != "undefined") {
363         // IE 9 and other non-IE browsers
364         sel = window.getSelection();
365 
366         // Test that the Selection object contains at least one Range
367         if (sel.getRangeAt && sel.rangeCount) {
368             // Get the first Range (only Firefox supports more than one)
369             range = window.getSelection().getRangeAt(0);
370             range.deleteContents();
371 
372             // Create a DocumentFragment to insert and populate it with HTML
373             // Need to test for the existence of range.createContextualFragment
374             // because it's non-standard and IE 9 does not support it
375             if (range.createContextualFragment) {
376                 fragment = range.createContextualFragment(html);
377             } else {
378                 // In IE 9 we need to use innerHTML of a temporary element
379                 var div = document.createElement("div"), child;
380                 div.innerHTML = html;
381                 fragment = document.createDocumentFragment();
382                 while ((child = div.firstChild)) {
383                     fragment.appendChild(child);
384                 }
385             }
386             var firstInsertedNode = fragment.firstChild;
387             var lastInsertedNode = fragment.lastChild;
388             range.insertNode(fragment);
389             if (selectInserted) {
390                 if (firstInsertedNode) {
391                     range.setStartBefore(firstInsertedNode);
392                     range.setEndAfter(lastInsertedNode);
393                 }
394                 sel.removeAllRanges();
395                 sel.addRange(range);
396             }
397         }
398     } else if (document.selection && document.selection.type != "Control") {
399         // IE 8 and below
400         range = document.selection.createRange();
401         range.pasteHTML(html);
402     }
403 }
